/**
 * Punto de entrada GET: devuelve el JSON con todos los datos del dashboard.
 * URL de uso: deploy como "Web App" → ejecutar como "Yo" → acceso "Cualquiera"
 */
function doGet(e) {
  const data = buildDashboardData();
  const json = JSON.stringify(data);
  const callback = e && e.parameter && e.parameter.callback;
  // JSONP: evita CORS desde GitHub Pages u otros dominios externos
  if (callback && /^[a-zA-Z_$][\w$]*$/.test(callback)) {
    return ContentService
      .createTextOutput(callback + '(' + json + ')')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  return ContentService
    .createTextOutput(json)
    .setMimeType(ContentService.MimeType.JSON);
}

/**
 * Construye el objeto completo de datos del dashboard leyendo el Sheet.
 */
function buildDashboardData() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const ingSheet = getSheetByNames_(ss, ['Ingresos2', 'Ingresos']);
  const gasSheet = getSheetByNames_(ss, ['Gastos2', 'Gastos']);
  const compendio = buildCompendioTotales_(ss);

  return {
    ingSpendee:            buildGroupedByMonthKeySheet(ingSheet, 'I', 'C', 'N'),
    ingPresup:             buildGroupedByMonthKeySheet(ingSheet, 'I', 'C', 'L'),
    gastoSpendee:          buildGroupedByMonthKeySheet(gasSheet, 'I', 'C', 'N'),
    gastoPresup:           buildGroupedByMonthKeySheet(gasSheet, 'I', 'C', 'L'),
    gastoByConcepto:       buildConceptoTotalsSheet(gasSheet),
    gastoByConceptoDetail: buildConceptoByGrupoMonthSheet(gasSheet),
    gastoByConceptoMonth:  buildConceptoByMonthSheet(gasSheet),
    // Vista General — Compendio General: Ingresos = col L, Gastos = col N
    ingCompendio:          compendio.ingByGrupo,
    ingCompendioMonth:     compendio.ingByMonth,
    gastoCompendio:        compendio.gasByGrupo,
    gastoCompendioMonth:   compendio.gasByMonth,
    estadosFinancieros:    buildEstadosFinancieros(ss)
  };
}

/**
 * Compendio General (Vista General):
 *   C = Mes y Año key | I = Grupo
 *   L = Ingresos  → Total Ingresos
 *   N = Gastos    → Total Gastos
 * No se filtra por Tipo: L e N son columnas distintas de monto.
 */
function buildCompendioTotales_(ss) {
  const sheet = getSheetByNames_(ss, ['Compendio General', 'Compendio']);
  const empty = { ingByGrupo: {}, ingByMonth: {}, gasByGrupo: {}, gasByMonth: {} };
  if (!sheet) return empty;

  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return empty;

  const keyCols = getSheetKeyColumns_(sheet, 'C');
  const grupoIdx = findColByHeader_(sheet, ['Grupo']) || colLetterToIndex('I');
  const ingIdx = colLetterToIndex('L');  // Ingresos
  const gasIdx = colLetterToIndex('N');  // Gastos
  const maxCol = Math.max(keyCols.mes, keyCols.anio, keyCols.key, grupoIdx, ingIdx, gasIdx);
  const data = sheet.getRange(2, 1, lastRow, maxCol).getValues();

  const ingByGrupo = {};
  const ingByMonth = {};
  const gasByGrupo = {};
  const gasByMonth = {};

  data.forEach(function(row) {
    const grupo = String(row[grupoIdx - 1] || '').trim() || 'Sin grupo';
    const key = rowMonthKey_(row, keyCols);
    if (!key) return;

    const ingreso = parseFloat(row[ingIdx - 1]) || 0;
    const gasto = parseFloat(row[gasIdx - 1]) || 0;

    if (ingreso !== 0) {
      if (!ingByGrupo[grupo]) ingByGrupo[grupo] = {};
      ingByGrupo[grupo][key] = round2((ingByGrupo[grupo][key] || 0) + ingreso);
      ingByMonth[key] = round2((ingByMonth[key] || 0) + ingreso);
    }
    if (gasto !== 0) {
      if (!gasByGrupo[grupo]) gasByGrupo[grupo] = {};
      gasByGrupo[grupo][key] = round2((gasByGrupo[grupo][key] || 0) + gasto);
      gasByMonth[key] = round2((gasByMonth[key] || 0) + gasto);
    }
  });

  return {
    ingByGrupo: ingByGrupo,
    ingByMonth: ingByMonth,
    gasByGrupo: gasByGrupo,
    gasByMonth: gasByMonth
  };
}

/**
 * Lee una hoja y agrupa valores por [Grupo][MesAñoKey].
 * @param {Spreadsheet} ss
 * @param {string} sheetName  Nombre de la hoja
 * @param {string} grupoCol   Letra de la columna Grupo   (ej: 'I')
 * @param {string} keyCol     Letra de la columna MesAño  (ej: 'C')
 * @param {string} valorCol   Letra de la columna Valor   (ej: 'N' = Spendee, 'L' = Presup.)
 * @returns {Object}  { "Marcas": { "1-2024": 1234.56, ... }, ... }
 */
function buildGroupedByMonthKey(ss, sheetName, grupoCol, keyCol, valorCol) {
  return buildGroupedByMonthKeySheet(getSheetByNames_(ss, [sheetName]), grupoCol, keyCol, valorCol);
}

function buildGroupedByMonthKeySheet(sheet, grupoCol, keyCol, valorCol) {
  if (!sheet) return {};

  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return {};

  var gIdx, vIdx, keyCols;
  if (/^Gastos/i.test(sheet.getName())) {
    var cols = getGastosColumnIndexes_(sheet);
    gIdx = cols.grupo;
    vIdx = String(valorCol || '').toUpperCase() === 'L' ? findPresupuestoCol_(sheet) : colLetterToIndex('N');
    keyCols = { mes: cols.mes, anio: cols.anio, key: cols.key };
  } else {
    keyCols = getSheetKeyColumns_(sheet, keyCol);
    gIdx = findColByHeader_(sheet, ['Grupo']) || colLetterToIndex(grupoCol);
    vIdx = findValorCol_(sheet, valorCol);
  }

  const maxCol = Math.max(gIdx, keyCols.mes, keyCols.anio, keyCols.key, vIdx);
  // Incluir última fila (antes se usaba lastRow-1 y se perdían gastos recientes)
  const data = sheet.getRange(2, 1, lastRow, maxCol).getValues();
  const result = {};

  data.forEach(function(row) {
    const grupo = String(row[gIdx - 1] || '').trim();
    const key   = rowMonthKey_(row, keyCols);
    const valor = parseFloat(row[vIdx - 1]) || 0;
    if (!grupo || !key || valor === 0) return;
    if (!result[grupo]) result[grupo] = {};
    result[grupo][key] = round2((result[grupo][key] || 0) + valor);
  });

  return result;
}

/**
 * Columnas en hoja Gastos/Gastos2:
 *   Grupo → col I | Mes y Año - key → col C | Concepto → col J | Spendee → col N
 */
function getGastosColumnIndexes_(sheet) {
  var keyCols = getSheetKeyColumns_(sheet, 'C');
  return {
    mes: keyCols.mes,
    anio: keyCols.anio,
    key: keyCols.key,
    grupo: findColByHeader_(sheet, ['Grupo']) || colLetterToIndex('I'),
    concepto: findColByHeader_(sheet, ['Concepto']) || colLetterToIndex('J'),
    valor: findSpendeeCol_(sheet)
  };
}

function findSpendeeCol_(sheet) {
  var byHeader = findColByHeader_(sheet, ['Spendee']);
  return byHeader || colLetterToIndex('N');
}

function findPresupuestoCol_(sheet) {
  return findColByHeader_(sheet, [
    'Presupuesto en cantidades', 'Presupuesto', 'Presup.', 'Presupuest', 'Presupuestc'
  ]) || colLetterToIndex('L');
}

function buildConceptoByGrupoMonth(ss, sheetName) {
  return buildConceptoByGrupoMonthSheet(getSheetByNames_(ss, [sheetName, 'Gastos']));
}

function buildConceptoByGrupoMonthSheet(sheet) {
  if (!sheet) return {};

  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return {};

  const cols = getGastosColumnIndexes_(sheet);
  const maxCol = Math.max(cols.grupo, cols.mes, cols.anio, cols.key, cols.concepto, cols.valor);
  const data = sheet.getRange(2, 1, lastRow, maxCol).getValues();
  const result = {};
  const keyCols = { mes: cols.mes, anio: cols.anio, key: cols.key };

  data.forEach(function(row) {
    const grupo    = String(row[cols.grupo - 1] || '').trim();
    const key      = rowMonthKey_(row, keyCols);
    const concepto = String(row[cols.concepto - 1] || '').trim();
    const valor    = parseFloat(row[cols.valor - 1]) || 0;
    if (!grupo || !key || !concepto || valor === 0) return;
    if (!result[grupo]) result[grupo] = {};
    if (!result[grupo][key]) result[grupo][key] = {};
    result[grupo][key][concepto] = round2((result[grupo][key][concepto] || 0) + valor);
  });

  return result;
}

/** { "1-2024": { "Publicidad": 100, "Software": 50 } } — etiquetas = Concepto */
function buildConceptoByMonthSheet(sheet) {
  const detail = buildConceptoByGrupoMonthSheet(sheet);
  const result = {};

  Object.keys(detail).forEach(function(grupo) {
    Object.keys(detail[grupo]).forEach(function(key) {
      if (!result[key]) result[key] = {};
      Object.keys(detail[grupo][key]).forEach(function(concepto) {
        result[key][concepto] = round2((result[key][concepto] || 0) + detail[grupo][key][concepto]);
      });
    });
  });

  return result;
}

/**
 * Totales por concepto (all-time). Derivado de buildConceptoByGrupoMonth.
 */
function buildConceptoTotals(ss, sheetName) {
  return buildConceptoTotalsSheet(getSheetByNames_(ss, [sheetName, 'Gastos']));
}

function buildConceptoTotalsSheet(sheet) {
  const detail = buildConceptoByGrupoMonthSheet(sheet);
  const result = {};

  Object.keys(detail).forEach(function(grupo) {
    Object.keys(detail[grupo]).forEach(function(key) {
      Object.keys(detail[grupo][key]).forEach(function(concepto) {
        result[concepto] = round2((result[concepto] || 0) + detail[grupo][key][concepto]);
      });
    });
  });

  const sorted = {};
  Object.entries(result)
    .sort((a, b) => b[1] - a[1])
    .forEach(([k, v]) => { sorted[k] = v; });

  return sorted;
}

/**
 * Lee la hoja "Estados Financieros" y devuelve el objeto:
 * { "3-2025": { ingreso: X, gasto: Y, utilidad: Z }, ... }
 *
 * Estructura de la hoja:
 *   Col C = Mes y Año key  (ej: "3-2025")
 *   Col F = Tipo           ("Ingreso" | "Gasto" | "Utilidad")
 *   Col G = Monto
 */
function buildEstadosFinancieros(ss) {
  const sheet = ss.getSheetByName('Estados Financieros');
  if (!sheet) return {};

  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return {};

  const cIdx = colLetterToIndex('C'); // Mes y Año key
  const fIdx = colLetterToIndex('F'); // Tipo
  const gIdx = colLetterToIndex('G'); // Monto
  const maxCol = Math.max(cIdx, fIdx, gIdx);

  const data = sheet.getRange(2, 1, lastRow, maxCol).getValues();
  const result = {};

  data.forEach(function(row) {
    const key   = normalizeMonthKey_(row[cIdx - 1]);
    const tipo  = String(row[fIdx - 1] || '').trim().toLowerCase();
    const monto = parseFloat(row[gIdx - 1]) || 0;

    if (!key || !tipo || monto === 0) return;
    if (!result[key]) result[key] = { ingreso: 0, gasto: 0, utilidad: 0 };

    if (tipo === 'ingreso')   result[key].ingreso   = round2(monto);
    else if (tipo === 'gasto')    result[key].gasto    = round2(monto);
    else if (tipo === 'utilidad') result[key].utilidad  = round2(monto);
  });

  return result;
}

/** Convierte letra(s) de columna a número (A=1, Z=26, AA=27…) */
function colLetterToIndex(col) {
  col = col.toUpperCase();
  let n = 0;
  for (let i = 0; i < col.length; i++) {
    n = n * 26 + (col.charCodeAt(i) - 64);
  }
  return n;
}

function getSheetByNames_(ss, names) {
  for (var i = 0; i < names.length; i++) {
    var sheet = ss.getSheetByName(names[i]);
    if (sheet) return sheet;
  }
  return null;
}

function findColByHeader_(sheet, headerNames) {
  if (!sheet) return null;
  var lastCol = sheet.getLastColumn();
  if (lastCol < 1) return null;
  var headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0];
  var want = headerNames.map(function(h) { return String(h).trim().toLowerCase(); });
  for (var i = 0; i < headers.length; i++) {
    var h = String(headers[i] || '').trim().toLowerCase();
    if (want.indexOf(h) >= 0) return i + 1;
  }
  return null;
}

/** Spendee (N) y Presupuesto (L) están separados — no mezclar al buscar columna */
function findValorCol_(sheet, valorCol) {
  var letter = String(valorCol || '').toUpperCase();
  if (letter === 'N') return findSpendeeCol_(sheet);
  if (letter === 'L') return findPresupuestoCol_(sheet);
  return colLetterToIndex(letter) || findSpendeeCol_(sheet);
}

/** Columnas A/B/C: Año - key, Mes - key, Mes y Año - key → siempre "M-YYYY" (ej. 1-2024) */
function getSheetKeyColumns_(sheet, keyColFallback) {
  return {
    mes: findColByHeader_(sheet, ['Mes - key', 'Mes key']) || colLetterToIndex('B'),
    anio: findColByHeader_(sheet, ['Año - key', 'Año key']) || colLetterToIndex('A'),
    key: findColByHeader_(sheet, ['Mes y Año - key', 'Mes y Año key', 'Mes y Año']) || colLetterToIndex(keyColFallback || 'C')
  };
}

function normalizeMonthKey_(raw, mesCell, anioCell) {
  if (raw instanceof Date && !isNaN(raw.getTime())) {
    return (raw.getMonth() + 1) + '-' + raw.getFullYear();
  }
  var s = String(raw == null ? '' : raw).trim();
  var m = 0;
  var y = 0;
  var match = s.match(/^(\d{1,2})\s*[-/]\s*(\d{4})$/);
  if (match) {
    m = parseInt(match[1], 10);
    y = parseInt(match[2], 10);
  }
  if ((!m || !y) && mesCell != null && anioCell != null) {
    m = parseInt(String(mesCell).trim(), 10);
    y = parseInt(String(anioCell).trim(), 10);
  }
  if (!m || !y || m < 1 || m > 12) return '';
  return m + '-' + y;
}

function rowMonthKey_(row, keyCols) {
  return normalizeMonthKey_(
    row[keyCols.key - 1],
    row[keyCols.mes - 1],
    row[keyCols.anio - 1]
  );
}

function round2(n) {
  return Math.round(n * 100) / 100;
}

function onOpen() {
  SpreadsheetApp.getUi()
    .createMenu('📊 Dashboard')
    .addItem('Previsualizar datos JSON', 'previewData')
    .addToUi();
}

/** Ejecuta buildDashboardData y muestra un resumen en el log */
function previewData() {
  const data = buildDashboardData();
  const grupos = Object.keys(data.ingSpendee);
  const meses  = Object.keys(data.estadosFinancieros);
  const conceptos = Object.keys(data.gastoByConcepto).length;

  const msg =
    '✅ Datos listos\n' +
    'Grupos ingresos: ' + grupos.join(', ') + '\n' +
    'Meses EF: ' + meses.length + ' (' + meses[0] + ' → ' + meses[meses.length-1] + ')\n' +
    'Conceptos gasto: ' + conceptos + '\n' +
    'Meses Compendio ingresos: ' + Object.keys(data.ingCompendioMonth || {}).length + '\n' +
    'Meses Compendio gastos: ' + Object.keys(data.gastoCompendioMonth || {}).length;

  Logger.log(JSON.stringify(data, null, 2));
  SpreadsheetApp.getUi().alert('📊 Dashboard Data', msg, SpreadsheetApp.getUi().ButtonSet.OK);
}
